333 {
334  // Invoke the visit method for this node.
335  if (!(instance->*visitMethod)(node))
336  return;
337 
338  // If this node has a model with a mesh skin, visit the joint hierarchy within it
339  // since we don't add joint hierarchies directly to the scene. If joints are never
340  // visited, it's possible that nodes embedded within the joint hierarchy that contain
341  // models will never get visited (and therefore never get drawn).
342  Model* model = dynamic_cast<Model*>(node->getDrawable());
